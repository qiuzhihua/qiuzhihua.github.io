import "AccountInfo.h"

@implementation AccountInfo

  • (instancetype)init{

    self = [super init]; if (self) {

    NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
 _isLogin = [defaults boolForKey:@"logined"];

    } return self; }

//创建个单线程

  • (instancetype)shareAccountInfo{ //定义个单线程

    static AccountInfo *sharedAccount;

    static dispatch_once_t onceToken; dispatch_once(&onceToken,^{ sharedAccount= [[AccountInfo alloc] init]; }); return sharedAccount; }

//保存用户登录信息

  • (void)saveLoginInfo:(NSDictionary *)info{ }

//删除用户登录信息

  • (void)deleteAccountInfo{ NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults]; [defaults setObject:nil forKey:@"logined"]; [defaults synchronize]; }

//判断用户登录状态

  • (BOOL)logined{ return _isLogin; }

@end
